-- ─── 2a. update_updated_at_column ─────────────────────────────
-- Automatically updates the updated_at column on every row update.
-- Used by triggers on all tables that have an updated_at field.
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
  NEW.updated_at = NOW();
  RETURN NEW;
END;
$$;

-- ─── 2b. user_has_org_access ──────────────────────────────────
-- Returns TRUE if the currently authenticated user belongs to
-- the given organisation.  Used by RLS policies.
-- This is a SECURITY DEFINER function — it runs with elevated
-- permissions to look up the profiles table.  Review carefully.
CREATE OR REPLACE FUNCTION user_has_org_access(target_org uuid)
RETURNS boolean
LANGUAGE sql
STABLE
SECURITY DEFINER
AS $$
  SELECT EXISTS (
    SELECT 1
    FROM   profiles
    WHERE  id              = auth.uid()
    AND    organisation_id = target_org
    AND    status          != 'suspended'
  );
$$;

-- ─── 2c. user_org_id ──────────────────────────────────────────
-- Convenience: returns the organisation_id of the current user.
CREATE OR REPLACE FUNCTION user_org_id()
RETURNS uuid
LANGUAGE sql
STABLE
SECURITY DEFINER
AS $$
  SELECT organisation_id
  FROM   profiles
  WHERE  id = auth.uid()
  LIMIT  1;
$$;

-- ─── 2e. generate_pairing_code (OPTIONAL) ─────────────────────
-- Generates a random pairing code in TS-XXXX-XXXX format.
-- Can be used server-side to issue codes securely.
-- (Pairing codes are currently generated in the frontend demo;
--  this function supports server-side generation in Run 9+.)
CREATE OR REPLACE FUNCTION generate_pairing_code()
RETURNS text
LANGUAGE sql
AS $$
  SELECT 'TS-'
    || upper(substring(md5(random()::text) for 4))
    || '-'
    || upper(substring(md5(random()::text) from 5 for 4));
$$;

-- ─── 3.01  organisations ──────────────────────────────────────
-- Top-level tenant.  Each TrustSheild OS™ client is one org.
CREATE TABLE IF NOT EXISTS organisations (
  id           uuid        PRIMARY KEY DEFAULT uuid_generate_v4(),
  name         text        NOT NULL,
  slug         text        NOT NULL UNIQUE,
  plan_status  text        NOT NULL DEFAULT 'trial'
                           CHECK (plan_status IN ('trial','active','suspended','cancelled')),
  mode         text        NOT NULL DEFAULT 'live'
                           CHECK (mode IN ('demo','live')),
  created_at   timestamptz NOT NULL DEFAULT NOW(),
  updated_at   timestamptz NOT NULL DEFAULT NOW()
);
COMMENT ON TABLE organisations IS
  'Top-level tenant/client account for TrustSheild OS™.  One row per client.';

-- ─── 3.02  profiles ───────────────────────────────────────────
-- Dashboard user profiles, linked to Supabase auth.users.
-- role: owner | admin | analyst | responder | viewer
CREATE TABLE IF NOT EXISTS profiles (
  id               uuid        PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  organisation_id  uuid        REFERENCES organisations(id) ON DELETE CASCADE,
  display_name     text,
  role             text        NOT NULL DEFAULT 'viewer'
                               CHECK (role IN ('owner','admin','analyst','responder','viewer')),
  email            text,
  status           text        NOT NULL DEFAULT 'active'
                               CHECK (status IN ('active','suspended','deactivated')),
  created_at       timestamptz NOT NULL DEFAULT NOW(),
  updated_at       timestamptz NOT NULL DEFAULT NOW()
);
COMMENT ON TABLE profiles IS
  'Dashboard user profiles.  Linked 1:1 to Supabase auth.users.';

-- ─── 3.03  tracked_entities ───────────────────────────────────
-- People, companies, brands, products, domains, or public figures
-- that TrustSheild OS™ is monitoring on behalf of a client.
--
-- ⚠  IMPORTANT ETHICAL NOTE:
-- authorisation_scope MUST be set to one of:
--   owned_brand | client_approved | public_information |
--   internal_organisation | authorisation_missing
-- Monitoring with scope 'authorisation_missing' should be blocked
-- at the application layer and must NOT proceed to active scanning.
CREATE TABLE IF NOT EXISTS tracked_entities (
  id                   uuid        PRIMARY KEY DEFAULT uuid_generate_v4(),
  organisation_id      uuid        NOT NULL REFERENCES organisations(id) ON DELETE CASCADE,
  entity_type          text        NOT NULL
                                   CHECK (entity_type IN (
                                     'Company','Brand','Person','Project',
                                     'Product','Domain','Public Figure','Organisation'
                                   )),
  display_name         text        NOT NULL,
  legal_or_public_name text,
  keywords             text[]      DEFAULT '{}',
  aliases              text[]      DEFAULT '{}',
  domains              text[]      DEFAULT '{}',
  public_urls          text[]      DEFAULT '{}',
  review_urls          text[]      DEFAULT '{}',
  region               text,
  authorisation_scope  text        NOT NULL DEFAULT 'authorisation_missing'
                                   CHECK (authorisation_scope IN (
                                     'owned_brand','client_approved',
                                     'public_information','internal_organisation',
                                     'authorisation_missing'
                                   )),
  monitoring_status    text        NOT NULL DEFAULT 'paused'
                                   CHECK (monitoring_status IN (
                                     'active','paused','archived','blocked'
                                   )),
  alert_sensitivity    text        NOT NULL DEFAULT 'medium'
                                   CHECK (alert_sensitivity IN (
                                     'low','medium','high','critical'
                                   )),
  notes                text,
  source               text        NOT NULL DEFAULT 'live'
                                   CHECK (source IN ('demo','live')),
  created_by           uuid        REFERENCES auth.users(id),
  created_at           timestamptz NOT NULL DEFAULT NOW(),
  updated_at           timestamptz NOT NULL DEFAULT NOW()
);
COMMENT ON TABLE tracked_entities IS
  'Entities (people/brands/companies/etc.) being monitored. '
  'authorisation_scope MUST be set — monitoring with scope '
  'authorisation_missing must be blocked at the app layer.';

-- SECTION 7 — RLS POLICIES
-- (Created AFTER RLS is enabled on each table)
--
-- POLICY DESIGN:
--   • Authenticated users can only access records belonging
--     to their own organisation (via user_has_org_access()).
--   • No anonymous (anon role) full-access policies exist.
--   • No broad "allow all" SELECT/INSERT/UPDATE policies exist.
--   • PWA direct write policies are restricted until secure
--     PWA authentication is completed in Run 9.
--   • audit_logs has SELECT + INSERT only — no UPDATE/DELETE.
--   • DELETE policies are omitted from most tables for safety;
--     add role-based deletes as needed in later runs.
--
-- NOTE ON PWA AUTHENTICATION:
--   Secure PWA auth (magic link, edge function tokens, or
--   backend-issued JWTs) is implemented in Run 9.
--   Until then, PWA reads/writes to Supabase must go through
--   authenticated dashboard users or a secure Edge Function.
--   Do NOT create anonymous public write policies for PWA tables.
-- ============================================================

-- ─── profiles ──────────────────────────────────────────────────
-- Users can read their own profile.
-- Users can read profiles in their organisation (for collaboration).
-- Users can update only their own profile.
CREATE POLICY "profiles_select_own"
  ON profiles FOR SELECT
  USING (id = auth.uid());

CREATE POLICY "profiles_select_org"
  ON profiles FOR SELECT
  USING (user_has_org_access(organisation_id));

CREATE POLICY "profiles_insert_own"
  ON profiles FOR INSERT
  WITH CHECK (id = auth.uid());

CREATE POLICY "profiles_update_own"
  ON profiles FOR UPDATE
  USING (id = auth.uid())
  WITH CHECK (id = auth.uid());

-- ─── organisations ─────────────────────────────────────────────
-- Members can read their own organisation.
-- Only owners/admins can update (enforced at app layer + here).
CREATE POLICY "organisations_select_own"
  ON organisations FOR SELECT
  USING (user_has_org_access(id));

CREATE POLICY "organisations_update_admin"
  ON organisations FOR UPDATE
  USING (
    EXISTS (
      SELECT 1 FROM profiles
      WHERE id = auth.uid()
      AND   organisation_id = organisations.id
      AND   role IN ('owner','admin')
    )
  );
